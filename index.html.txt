<script>
// Carrusel automático
let images = document.querySelectorAll(".carousel img");
let index = 0;

const params = new URLSearchParams(window.location.search);
let imgParam = parseInt(params.get("img"));
if (imgParam && images[imgParam-1]) index = imgParam-1;

function showImage() {
  images.forEach(img => img.classList.remove("active"));
  images[index].classList.add("active");
}
showImage();

setInterval(() => {
  index = (index + 1) % images.length;
  showImage();
}, 6000);

// Buscar iglesia
function buscarIglesia() {
  if (!navigator.geolocation) {
    window.open("https://www.google.com/maps/search/Iglesia+Adventista+del+Séptimo+Día");
    return;
  }
  navigator.geolocation.getCurrentPosition(pos => {
    let lat = pos.coords.latitude;
    let lon = pos.coords.longitude;
    window.open(`https://www.google.com/maps/search/Iglesia+Adventista+del+Séptimo+Día/@${lat},${lon},14z`);
  });
}
</script>
